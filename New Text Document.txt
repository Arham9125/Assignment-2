void main(List<String> args) {
  int marks = 70;
  {
    if (marks >= 90) 
{
      print("You got A1 grade");
    } 
else if (marks >= 80) 
{
      print("You got A grade");
    } 
else if (marks >= 70)
 {
      print("You got B grade");
    } 
else if (marks >= 60)
 {
      print("You got C grade");
    } 
else if (marks >= 50) 
{
      print("You got D grade");
    } 
else 
{
      print("You are fail");
    }
  }
}
